---

- Create `./webpack.config.js`

```javascript
const path = require('path');

module.exports = {
  entry: './src/index.js',
  output: {
    filename: 'main.js',
    path: path.resolve(__dirname, 'dist'),
  },
  resolve: {
    modules: [path.resolve(__dirname, 'src')],
  },
};
```

---

- create `./jsconfig.json`

```json
{
  "compilerOptions": {
    "baseUrl": "./src"
  },
  "include": ["src"],
  "exclude": ["node_modules"]
}
```
